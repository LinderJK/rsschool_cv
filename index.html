                    function changeClass(nElem, method, animation) {
                      if (method === "add") {
                        nElem.classList.add(animation);
                      } else {
                        nElem.classList.remove(animation);
                      }
                    }

                    // FIND_ALL_RES

                      return {
                        relem: resObj,
                        rinput: resInputObj,
